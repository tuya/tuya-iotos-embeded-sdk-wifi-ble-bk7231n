typedef struct mbedtls_asn1_buf
{
   int tag;               /**< ASN1 type, e.g. MBEDTLS_ASN1_UTF8_STRING. */
   size_t len;            /**< ASN1 length, in octets. */
   unsigned char *p;      /**< ASN1 data, e.g. in ASCII. */
