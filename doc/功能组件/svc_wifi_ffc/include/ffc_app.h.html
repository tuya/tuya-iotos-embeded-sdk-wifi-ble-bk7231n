Typedef ffc_cfg_t;


      
typedef struct {
   /** uuid,assigned by TUYA */
   uint8_t    uuid[25 + 1];
   /** pid,assigned by TUYA */
   uint8_t    pid[20 + 1];
   /** auzkey,assigned by TUYA */
   uint8_t    auzkey[32 + 1];
   /** mac address */
   uint8_t    mac[6];
   /** wifi channel */
   uint8_t    channel[13];
   /** channel count */
   uint8_t    channel_count;
   /** flag */
   uint32_t   flag;
} ffc_cfg_t;

typedef struct {
   /** new channel set */
   int (*channel_set)(uint8_t new_channel);

   uint8_t (*channel_get)(void);
