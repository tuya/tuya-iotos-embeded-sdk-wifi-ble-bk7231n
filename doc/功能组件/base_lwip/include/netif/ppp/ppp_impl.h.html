Marco GETCHAR(c, cp)

      
#define GETCHAR(c, cp) { \
 (c) = *(cp)++; \
}


      
/*
 * Inline versions of get/put char/short/long.
 * Pointer is advanced; we assume that both arguments
 * are lvalues and will already be in registers.
 * cp MUST be u_char *.
 */


      
Marco PUTCHAR(c, cp)

      
#define PUTCHAR(c, cp) { \
 *(cp)++ = (u_char) (c); \
}


      


      
Marco GETSHORT(s, cp)

      
#define GETSHORT(s, cp) { \
 (s) = *(cp)++ << 8; \
 (s) |= *(cp)++; \
}


      


      
Marco PUTSHORT(s, cp)

      
#define PUTSHORT(s, cp) { \
 *(cp)++ = (u_char) ((s) >> 8); \
 *(cp)++ = (u_char) (s); \
}


      


      
Marco GETLONG(l, cp)

      
#define GETLONG(l, cp) { \
 (l) = *(cp)++ << 8; \
 (l) |= *(cp)++; (l) <<= 8; \
 (l) |= *(cp)++; (l) <<= 8; \
 (l) |= *(cp)++; \
}


      


      
Marco PUTLONG(l, cp)

      
#define PUTLONG(l, cp) { \
 *(cp)++ = (u_char) ((l) >> 24); \
 *(cp)++ = (u_char) ((l) >> 16); \
 *(cp)++ = (u_char) ((l) >> 8); \
 *(cp)++ = (u_char) (l); \
}
