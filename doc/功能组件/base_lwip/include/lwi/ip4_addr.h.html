Marco ip4_addr_netcmp(addr1, addr2, mask)

      
#define ip4_addr_netcmp(addr1, addr2, mask) (((addr1)->addr & \
                                             (mask)->addr) == \
                                            ((addr2)->addr & \
                                             (mask)->addr))
