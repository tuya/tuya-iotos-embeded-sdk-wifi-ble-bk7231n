Marco FREE_LIST(type, p, list_name)\
{\

      
#define FREE_LIST(type, p, list_name)\
{\
   type *posnode;\
   while(!tuya_list_empty(&(p)->list_name)) {\
   posnode = tuya_list_entry((&(p)->list_name)->next, type, list_name);\
   tuya_list_del((&(p)->list_name)->next);\
   Free(posnode);\
   }\
}
