Marco tuya_spi_send_then_send(spi, send1, len1, send2, len2)

      
#define tuya_spi_send_then_send(spi, send1, len1, send2, len2)             \
   do {                                                                   \
       tuya_spi_msg_t   msg[2];                                          \
       msg[0].send_buf = send1;                                           \
       msg[0].recv_buf = NULL;                                            \
       msg[0].length  = len1;                                            \
       msg[1].send_buf = send2;                                           \
       msg[1].recv_buf = NULL;                                            \
       msg[1].length  = len2;                                            \
       tuya_spi_xfer_msg(spi, msg, 2);                                    \
   } while (0)
