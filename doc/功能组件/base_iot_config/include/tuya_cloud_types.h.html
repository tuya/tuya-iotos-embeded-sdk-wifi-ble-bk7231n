#define DDWORD_SWAP(X) { \
   unsigned long temp_low, temp_high; \
   temp_low = DWORD_SWAP((unsigned long)((X) & 4294967295)); \
   temp_high = DWORD_SWAP((unsigned long)((X) >> 32)); \
   X = 0; \
   X |= temp_low; \
   X <<= 32; \
   X |= temp_high;\
